
int array[10];
int mark[10];


void print_func(int n) {
	int j;
	for(j = 0; j < n; j = j + 1) {
		printf(array[j]);
	}
}


void func(int loc, int n) {
	int i;

	if (loc == n) {
		print_func(n);
		return;
	}

	for (i = 0; i < n; i = i + 1) {
		if (mark[i] == 0) {
			array[loc] = i + 1;
			mark[i] = 1;
			func(loc + 1, n);
			mark[i] = 0;
		}
	}
}


void main() {
	int i;
	int n;
	scanf(n);

	for (i = 0; i < 10; i = i + 1) {
		array[i] = 0;
		mark[i] = 0;
	}

	func(0, n);
	printf("aaaa\n");
}
